# la resolution de dependances du jour.
#

# l'environnement de build reellement utilise pour PyInstaller :
#
#   python -m pip install --upgrade -r requirements.txt
#   python -m pip freeze | findstr /i "cryptography cffi pycparser" > requirements-lock.txt
#
# Pour un build reproductible a partir de ce fichier :
#
#   python -m pip install -r requirements-lock.txt
